予約システム制作 MySQL テーブル設計
⑴テーブル名：users(親) / reservation(子)
⑵カラム
  shop :
      id INT NOT NULL AOUT_INCREMENT,
      shop_name varchar(128) NOT NULL,
      shop_id varchar(16) NOT NULL,
      email vvarchar(256) NOT NULL,
      password INT NOT NULL,
      reservable_date DATE NOT NUL,
      start_time TIME NOT NULL,
      end_time TIME NOT NULL,
      max_reserve_num INT NOT NULL,
      PRIMARY KEY (id)

  reserve :
      id INT NOT NULL AOUT_INCREMENT,
      reserve_date DATE NOT NULL,
      reserve_time TIME NOT NULL,
      reserve_num INT NOT NULL,
      name varchar(100) NOT NULL,
      email varchar(254) NOT NULL,
      tel varchar(20) NOT NULL,
      comment vmediumtext,
      created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
      updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
      PRIMARY KEY (id)
